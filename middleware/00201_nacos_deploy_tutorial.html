version: "3"

  mysql:
    hostname: mysql
    image: mysql:5.7.21
    container_name: mysql

      - n01
    volumes:
      - ./data/mysql:/var/lib/mysql
      - ./data/init.sql:/docker-entrypoint-initdb.d/init.sql:ro
    ports:
      - "3306:3306"
    environment:
      - MYSQL_ROOT_PASSWORD=root
      - MYSQL_DATABASE=nacos
    logging:
      driver: json-file
      options:
        max-size: 100m
        max-file: "3"
    restart: on-failure

  nacos:
    hostname: nacos
    image: nacos/nacos-server:v2.2.3

      - NACOS_AUTH_IDENTITY_VALUE=security
    logging:
      driver: json-file
      options:
        max-size: 100m
        max-file: "3"
    restart: on-failure
    depends_on:
      - mysql

networks:
