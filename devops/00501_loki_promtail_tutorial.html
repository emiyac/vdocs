TIP

loki配置 ​

data/loki/conf/loki.yml

yaml
auth_enabled: false    # 是否开启认证

server:
  http_listen_port: 3100    # 监听的端口
  grpc_server_max_recv_msg_size: 104857600    # 配置最大接受和发送数据的小100m
  grpc_server_max_send_msg_size: 104857600

common:
  ring:    # 配置hash环，用于将数据分配给不同的loki节点
    instance_addr: 127.0.0.1
    kvstore:
      store: inmemory    # 使用内存来存储hash环分配的loki节点信息
  replication_factor: 1
  path_prefix: /loki    # 路径前缀

schema_config:
  configs:
  - from: 2020-05-15    # 配置索引和存储的起始时间
    store: tsdb    # 存储引擎使用tsdb，boltdb-shipper在新版本已经废弃
    object_store: filesystem    # 对象存储使用文件系统存储
    schema: v13
    index:
      prefix: index_    # 配置索引文件的前缀为index_
      period: 24h    # 配置索引文件的周期为24h

storage_config:
  filesystem:
    directory: /loki/chunks    # 指定本地文件系统存储路径

limits_config:
  reject_old_samples: true    # 是否拒绝旧日志
  reject_old_samples_max_age: 168h    # 配置拒绝旧日志的时间为168h
  ingestion_rate_mb: 16    # 配置日志的接收速率16m/s
  ingestion_burst_size_mb: 32    # 配置日志接收的桶的大小为32m
  max_entries_limit_per_query: 2000    # 配置查询的最大条目数

chunk_store_config:
  max_look_back_period: 168h    # 可查看日志的最大回溯时间

table_manager:
  retention_deletes_enabled: true   # 保留删除开启
  retention_period: 168h  # 超过72h的块数据将被删除

ruler:
  storage:
    type: local
    local:
      directory: /loki/rules    # 告警规则存储路径，在该目录的fake目录下
  rule_path: /loki/rules-temp
  alertmanager_url: http://alertmanager:9093    # alertmanager_url 地址
  ring:
    kvstore:
      store: inmemory
  enable_api: true
  enable_alertmanager_v2: true

alertmanager配置 ​

data/alertmanager/conf/alertmanager.yml

yaml
global:
  resolve_timeout: 1m    # 判断告警恢复时间
  smtp_from: "xxx@qq.com"
  smtp_smarthost: "smtp.qq.com:465"
  smtp_auth_username: "xxx@qq.com"
  smtp_auth_password: "xxx"
  smtp_require_tls: false
  smtp_hello: "qq.com"
route:
  group_by: ['alertname','job']
  # group_wait: 10s, group_interval: 3m; repeat_interval: 3m; test result: interval of recv email is 6m
  group_wait: 10s
  group_interval: 3m
  repeat_interval: 3m
  receiver: 'webhook'    # 默认发送的目标
  routes:
  - match:    # 通过匹配判断发送的目标
      level: info
    receiver: "email"
      
templates:
  - '/etc/alertmanager/templates/default.tmpl'

receivers:
- name: "email"
  email_configs:
  - to: "xxx@qq.com"
    send_resolved: false
    html: '{{ template "wechat.default.message" }}'

- name: 'webhook'
  webhook_configs:
  - send_resolved: true
    url: 'http://alertmanager-webhook:8000/dingtalk?tmpl=dingtalk01.tmpl'

⚡alertmanager时间相关的参数说明：alertmanager在收到一条新的告警之后，会等待 group_wait 时间，对这条新的告警做一些分组、更新、静默的操作。当第一条告警经过 group_wait 时间之后，alertmanager 会每隔 group_interval 时间检查一次这条告警，判断是否需要对这条告警进行一些操作，当 alertmanager 经过 n 次 group_interval 的检查后，n * group_interval 恰好大于 repeat_interval 的时候，alertmanager 才会将这条告警再次发送给对应的 receiver

yaml
group_wait: 10s
group_interval: 3m
repeat_interval: 3m

docker-compose配置 ​

yaml
version: "3"

services:
  loki:
    container_name: loki
    hostname: loki
    image: grafana/loki:2.9.4
    ports:
      - 3100:3100
    volumes:
      - ./data/loki/conf:/etc/loki
      - ./data/loki/data:/loki
    command: -config.file=/etc/loki/loki.yml
    restart: on-failure
    logging:
      driver: json-file
      options:
        max-size: 100m
        max-file: "3"
    networks:
      - lpg

  promtail:
    container_name: promtail
    hostname: promtail
    image: grafana/promtail:2.9.4
    volumes:
      - ./data/promtail/conf:/etc/promtail
      - ./data/promtail/data:/data
      - ./data/logs:/var/prod/logs    # promtail读取日志文件位置，看实际情况配置
    command: -config.file=/etc/promtail/promtail.yml
    restart: on-failure
    depends_on:
      - loki
    logging:
      driver: json-file
      options:
        max-size: 100m
        max-file: "3"
    networks:
      - lpg

  grafana:
    container_name: grafana
    hostname: grafana
    image: grafana/grafana:10.4.1
    restart: on-failure
    volumes:
      - ./data/grafana/data:/var/lib/grafana
    ports:
      - 3000:3000
    logging:
      driver: json-file
      options:
        max-size: 100m
        max-file: "3"
    networks:
      - lpg
      
  alertmanager:
    container_name: alertmanager
    hostname: alertmanager
    image: prom/alertmanager
    restart: on-failure
    volumes:
      - ./data/alertmanager/conf:/etc/alertmanager
    ports:
      - 9093:9093
    logging:
      driver: json-file
      options:
        max-size: 100m
        max-file: "3"
    networks:
      - lpg    
     
  # 基于fastapi开发的webhook插件，配置参考https://gitee.com/emica/alertmanager-webhook.git   
  alertmanager-webhook:
    container_name: alertmanager-webhook
    hostname: alertmanager-webhook
    image: alertmanager-webhook:v1
    ports:
      - 8000:8000
    environment:
      - GRAFANA_URL=xxx
      - WECHAT_WEBHOOK_KEY=xxx
      - WECHAT_TEMPLATE=wechat.tmpl
      - DINGTALK_WEBHOOK_ACCESS_TOKEN=xxx
      - DINGTALK_WEBHOOK_SECRET=xxx
      - DINGTALK_TEMPLATE=dingtalk.tmpl
      - DEBUG=true
    logging:
      driver: json-file
      options:
        max-size: 100m
        max-file: "3"
    networks:
      - lpg
networks:
  lpg:
    ipam:
      driver: default # default is bridge
      config:
        - subnet: 172.30.0.0/16
